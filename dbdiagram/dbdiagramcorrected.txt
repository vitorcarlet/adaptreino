Table athletes_notes {
  id integer [primary key]
  athlete_id integer
  desc string
}

Table athletes {
  id integer [primary key]
  name varchar
  age integer
  photo string
  created_at timestamp
}

Table sports {
  id integer [primary key]
  name varchar
}

Table training_models {
  id integer [primary key]
  name varchar
  description text
  sport_id integer
}

Table training_model_fields {
  id integer [primary key]
  training_model_id integer
  schope varchar // dizer se o campo do formulario Ã© do
  field_name varchar // A descriptive name, e.g., "Heart Rate", "Perceived Exertion"
  field_type varchar // 'text', 'number', 'boolean', 'single_choice', 'multiple_choice', etc.
  sort_order integer
  desactivated boolean
  isChild boolean
  dadId integer
  data jsonb
}

Table trainings {
  id integer [primary key]
  user_id integer // Corresponds to an athlete
  date timestamp
  notes text      // General notes for the entire training session
  model_id integer [null]
  sport_id integer
  data jsonb [null]
  created_at timestamp
  updated_at timestamp
}


// --------------------------------------------------

Ref: athletes_notes.athlete_id > athletes.id
